Вместо 
const getAbs = (num) => {
  if (num >= 0) {
    return num;
  }
  return -num;
};

Пример1
const factorial = (n) => {
    if (n===1) {
        return 1;
    }
    else {
        return n*factorial(n-1);
    }
}
const answer = factorial(3);
